#Mesh parameters
   subsection Mesh
	set Number of spatial refinements = 3
	set Polynomial Degree = 3
   end 
#Physical parameters
   subsection Physics
	set Mass diffusion = 8e-2
	set Vorticity diffusion = 8e-2
   end 
#Runtime
   subsection Time
	set Timestep size dt = 0.00100
	set Number of timesteps = 12000 #2000000
	set Hard endtime = 15000.0
	set fstep = 0
	set RK order = 2
	set RK stage = 2 # >3 for RKC
	set Method = 1 # 1 = explicit, 2 = IMEX
	set RK type = 1 # 1 = RKSSP, 2 = RKC
	set eps = 0.153846154 # RKC-epsilon
	set Output_modulus = 100 #3000
   end
#Regularizating parameters
   subsection Regularity
	set Artificial diffusion = false
	set epsilon weight = 1.e-1
   end 
#Convergence
   subsection Convergence
	set Threshold = 1.0e-4
	set Iterations = 10000
	set Fast Threshold = 1e-10
	set Fast Iterations = 50
   end
#Loading/Saving
   subsection EllipticSolver	
        set Penalty type = 1 # SIPG=1,NIPG=-1,IIPG=0
        set Sigma penalty = 50.0 #75.0  #200.0
   end
#Loading/Saving
   subsection SavedSolution
	set Load = false
	set Save = true
	set h    = 3
        set p    = 3
   end
